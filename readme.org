# logging_config.py
from loguru import logger
from pathlib import Path
from datetime import datetime
import sys

LOG_ROOT = Path("logs")

def setup_logging():
    logger.remove()  # remove default handler

    # Console logging (nice during development)
    logger.add(
        sys.stdout,
        level="INFO",
        format="<green>{time:YYYY-MM-DD HH:mm:ss.SSS}</green> | "
               "<level>{level: <8}</level> | "
               "<cyan>{name}</cyan>:<cyan>{function}</cyan>:<cyan>{line}</cyan> - "
               "<level>{message}</level>",
        enqueue=True,  # important for async environments
        backtrace=True,
        diagnose=False,
    )

    # Application file logging (daily)
    app_log_dir = LOG_ROOT / "application"
    app_log_dir.mkdir(parents=True, exist_ok=True)

    logger.add(
        app_log_dir / "{time:YYYY-MM-DD}.log",
        level="INFO",
        rotation="00:00",           # rotate daily at midnight
        retention="30 days",        # keep 30 days
        compression="zip",          # compress old logs
        enqueue=True,
        format="{time:YYYY-MM-DD HH:mm:ss.SSS} | {level} | {extra} | {message}",
    )

4️⃣ WebSocket Logging Per req_id
Now the interesting part.

We dynamically create a sink per req_id.

Helper Function
# robot_logging.py
from loguru import logger
from pathlib import Path
from datetime import datetime

LOG_ROOT = Path("logs")


def get_robot_logger(req_id: str):
    today = datetime.now().strftime("%Y-%m-%d")

    robot_log_dir = LOG_ROOT / "robots" / today
    robot_log_dir.mkdir(parents=True, exist_ok=True)

    file_path = robot_log_dir / f"{req_id}.log"

    # Add sink specific to this req_id
    sink_id = logger.add(
        file_path,
        level="DEBUG",
        rotation="100 MB",
        retention="14 days",
        compression="zip",
        enqueue=True,
        format="{time:YYYY-MM-DD HH:mm:ss.SSS} | {level} | {message}",
    )

    # Return a bound logger + sink id (so we can clean up later)
    return logger.bind(req_id=req_id), sink_id
5️⃣ WebSocket Endpoint Example
from fastapi import WebSocket
from robot_logging import get_robot_logger
from loguru import logger

@app.websocket("/robots/ws")
async def robot_ws(websocket: WebSocket):
    req_id = websocket.query_params.get("req_id")

    if not req_id:
        await websocket.close(code=4000)
        return

    await websocket.accept()

    robot_logger, sink_id = get_robot_logger(req_id)

    robot_logger.info("WebSocket connection established")

    try:
        while True:
            data = await websocket.receive_text()
            robot_logger.debug(f"Received: {data}")
            await websocket.send_text("ok")

    except Exception as e:
        robot_logger.exception("WebSocket error")

    finally:
        robot_logger.info("WebSocket connection closed")
        logger.remove(sink_id)  # IMPORTANT cleanup
6️⃣ Important Best Practices for Production
✅ Always use enqueue=True
FastAPI is async → prevents race conditions in logging.

✅ Remove per-request sink on disconnect
If you don’t call logger.remove(sink_id) you will leak file handles.

✅ Consider log size limits
Long-running websocket connections can generate huge logs.

You might prefer:

rotation="50 MB"
retention="7 days"
✅ Consider structured JSON logs (optional but powerful)
Instead of text:

logger.add(
    file_path,
    serialize=True,  # JSON logs
)
Very useful if you later push logs to ELK / Loki.

7️⃣ Advanced Option: Automatic Filtering Instead of Manual Sink Creation
If you expect many concurrent connections, dynamically adding sinks may become heavy.

Alternative approach:

Use a single robot sink and filter per req_id:

logger.add(
    "logs/robots/{time:YYYY-MM-DD}/{extra[req_id]}.log",
    filter=lambda record: "req_id" in record["extra"],
)
Then simply:

robot_logger = logger.bind(req_id=req_id)
robot_logger.info("Something happened")
This avoids manually adding/removing sinks.

👉 This is usually the cleanest solution.

# logging_config.py
from loguru import logger
from logging_context import request_id_ctx
from pathlib import Path
import sys

LOG_ROOT = Path("logs")

def inject_request_id(record):
    record["extra"]["req_id"] = request_id_ctx.get()
    return record

def setup_logging():
    logger.remove()

    logger = logger.patch(inject_request_id)

    # Console
    logger.add(
        sys.stdout,
        level="INFO",
        enqueue=True,
        format="{time} | {level} | {extra[req_id]} | {message}",
    )

    # Application log
    logger.add(
        LOG_ROOT / "application/{time:YYYY-MM-DD}.log",
        rotation="00:00",
        retention="30 days",
        enqueue=True,
        format="{time} | {level} | {extra[req_id]} | {message}",
    )

    # Robot logs (automatic routing!)
    logger.add(
        LOG_ROOT / "robots/{time:YYYY-MM-DD}/{extra[req_id]}.log",
        filter=lambda r: r["extra"].get("req_id") is not None,
        rotation="100 MB",
        retention="14 days",
        enqueue=True,
        format="{time} | {level} | {message}",
    )

    return logger

When you run FastAPI with multiple workers (via Uvicorn or Gunicorn), each worker is:

a separate OS process

with its own memory

its own contextvars

contextvars propagate within a single process only.

They do NOT automatically propagate across workers.

🚨 The Core Rule
You cannot “share” a request context between workers.

Instead, you must:

Ensure the correct req_id is passed into the worker handling the request

Ensure each worker logs safely without corrupting shared log files

🧠 Case 1 — Standard Multi-Worker FastAPI
If you start with:

uvicorn main:app --workers 4
Each worker:

Handles its own WebSocket

Sets its own ContextVar

Writes its own logs

Your existing contextvars approach works perfectly per worker.

The real issue becomes:

Multiple workers writing to the same log file.

🔥 The Real Problem: File Contention
Loguru is thread-safe.

It is NOT multi-process safe by default.

If 4 workers write to:

logs/application/2026-02-27.log
you may get:

interleaved writes

corrupted rotation

broken compression
